# SGD
import numpy as np
import matplotlib.pyplot as plt

np.random.seed(123)

# Simulate data
n = 300
p = 2
X = np.random.randn(n, p)
beta_true = np.array([2.0, -3.0])
y = X @ beta_true + 0.5 * np.random.randn(n)

# Objective function
def loss(beta):
    residual = y - X @ beta
    return 0.5 * np.mean(residual ** 2)

# SGD
beta = np.zeros(p)
alpha0 = 0.2
n_epochs = 50
loss_history = []

for epoch in range(n_epochs):
    indices = np.random.permutation(n)

    for t, i in enumerate(indices):
        xi = X[i]
        yi = y[i]

        residual_i = yi - xi @ beta
        grad_i = -residual_i * xi

        alpha = alpha0 / (1 + 0.01 * (epoch * n + t))
        beta = beta - alpha * grad_i

    loss_history.append(loss(beta))

print("Estimated beta:", beta)
print("True beta:", beta_true)

plt.figure(figsize=(7, 4))
plt.plot(loss_history, marker='o')
plt.xlabel("Epoch")
plt.ylabel("Loss")
plt.title("SGD convergence for linear regression")
plt.grid(True)
plt.show()

# BFGS algorithm
import numpy as np
import matplotlib.pyplot as plt

# Rosenbrock function
def f(x):
    return (1 - x[0])**2 + 100 * (x[1] - x[0]**2)**2

# Gradient of Rosenbrock function
def grad_f(x):
    dfdx = -2 * (1 - x[0]) - 400 * x[0] * (x[1] - x[0]**2)
    dfdy = 200 * (x[1] - x[0]**2)
    return np.array([dfdx, dfdy])

# Backtracking line search using Armijo condition
def backtracking_line_search(x, p, grad, alpha0=1.0, rho=0.5, c=1e-4):
    alpha = alpha0
    fx = f(x)

    while f(x + alpha * p) > fx + c * alpha * np.dot(grad, p):
        alpha *= rho

    return alpha

# BFGS algorithm
def bfgs(x0, tol=1e-6, max_iter=1000):
    x = x0.astype(float)
    n = len(x)

    H = np.eye(n)
    path = [x.copy()]
    losses = [f(x)]

    for k in range(max_iter):
        g = grad_f(x)

        if np.linalg.norm(g) < tol:
            break

        p = -H @ g
        alpha = backtracking_line_search(x, p, g)

        x_new = x + alpha * p
        g_new = grad_f(x_new)

        s = x_new - x
        y = g_new - g
        ys = np.dot(y, s)

        # Curvature check
        if ys > 1e-12:
            rho = 1.0 / ys
            I = np.eye(n)

            H = (
                (I - rho * np.outer(s, y))
                @ H
                @ (I - rho * np.outer(y, s))
                + rho * np.outer(s, s)
            )
        else:
            # If curvature condition fails, reset H
            H = np.eye(n)

        x = x_new
        path.append(x.copy())
        losses.append(f(x))

    return x, np.array(path), np.array(losses)

# Initial value
x0 = np.array([-1.2, 1.0])

# Run BFGS
x_star, path, losses = bfgs(x0)

print("Estimated minimizer:", x_star)
print("Minimum value:", f(x_star))
print("Number of iterations:", len(losses) - 1)

# Plot loss convergence
plt.figure(figsize=(7, 4))
plt.semilogy(losses, marker='o')
plt.xlabel("Iteration")
plt.ylabel("Objective value: log scale")
plt.title("BFGS convergence on Rosenbrock function")
plt.grid(True)
plt.show()
