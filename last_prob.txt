#In the data, doing imputation, first do k-means clustering with 3 clusters. 
#Then, consider two classes, Setosa and Virginica. 
#Build a KNN with 5-fold cross validations. 
#Compare the results of KNN with the results of SVM on 5-folds, 
#by calculating the average sensitivity and accuracy

# Load Libraries
library(class)
library(caret)
library(e1071)

# Load Iris Dataset
data("iris")

View(iris)

# Imputation
for(i in 1:4){
  iris[,i][is.na(iris[,i])] <- mean(iris[,i], na.rm = TRUE)
}

# K-means Clustering with 3 Clusters
set.seed(123)

kmeans_result <- kmeans(iris[,1:4],
                        centers = 3)

kmeans_result$cluster

# Consider Only Setosa and Virginica
data2 <- subset(iris,
                Species %in% c("setosa",
                               "virginica"))

View(data2)

# KNN

# Build Training Data for KNN
train <- rbind(
  data2[1:25,-5],
  data2[51:75,-5]
)

# Build Test Data for KNN
test <- rbind(
  data2[26:50,-5],
  data2[76:100,-5]
)

# Labels
label <- c(
  rep("setosa",25),
  rep("virginica",25)
)

label <- as.factor(label)

# KNN Model
mod <- knn(train,
           test,
           label,
           k = 5)

mod

# Confusion Matrix for KNN
confusionMatrix(data = mod,
                reference = label,
                positive = "virginica")

# Table for KNN
table(mod,
      label)

# SVM

# Build Training Data for SVM
train <- rbind(
  data2[1:25,],
  data2[51:75,]
)

# Build Test Data for SVM
test <- rbind(
  data2[26:50,-5],
  data2[76:100,-5]
)

# SVM Model
mod <- svm(Species~.,
           train)

# Predict SVM
pred <- predict(mod,
                test)

# Match Levels
pred <- factor(pred,
               levels = levels(label))

pred

# Confusion Matrix for SVM
confusionMatrix(data = pred,
                reference = label,
                positive = "virginica")

# Table for SVM
table(pred,
      label)
